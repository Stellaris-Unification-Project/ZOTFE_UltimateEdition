create_ringworld_empire_fleet = {
	create_fleet = {
		name = "NAME_Gift_of_the_Ancestors"
		settings = {
			spawn_debris = no
			can_upgrade = no 
			can_disband = yes
			uses_naval_capacity = no
		}
		effect = {
			set_owner = ROOT.owner
			create_ringworld_empire_battlecruiser = yes
			while = {
				count = 3
				create_ringworld_empire_escort = yes
			}
			set_location = ROOT
		}
	}
}
